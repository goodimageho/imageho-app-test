async function cropAndSaveImages(sourceImage) {
  const img = new Image();
  img.src = sourceImage;
  
  await new Promise(resolve => img.onload = resolve);
  
  const crops = {
    header: { x: 0, y: 0, w: img.width, h: 50 },
    pink: { x: 0, y: 50, w: img.width, h: 80 },
    red: { x: 0, y: 130, w: img.width, h: 80 },
    yellow: { x: 0, y: 210, w: img.width, h: 80 },
    green: { x: 0, y: 290, w: img.width, h: 80 },
    blue: { x: 0, y: 370, w: img.width, h: 80 },
    checkboxLeft: { x: 0, y: 480, w: img.width/2, h: 250 },
    checkboxRight: { x: img.width/2, y: 480, w: img.width/2, h: 250 }
  };
  
  const croppedImages = {};
  
  for (const [name, crop] of Object.entries(crops)) {
    const canvas = document.createElement('canvas');
    canvas.width = crop.w;
    canvas.height = crop.h;
    const ctx = canvas.getContext('2d');
    ctx.drawImage(img, crop.x, crop.y, crop.w, crop.h, 0, 0, crop.w, crop.h);
    croppedImages[name] = canvas.toDataURL('image/png');
  }
  
  return croppedImages;
}
